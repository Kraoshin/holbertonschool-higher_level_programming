The 'example' module
====================

Using 'Sort'
------------


>>> my_list = MyList([3, 1, 2])
>>> my_list.print_sorted()
[1, 2, 3]

>>> my_list = MyList([])
>>> my_list.print_sorted()
[]

>>> my_list = MyList([-5, 10, 0, -3])
>>> my_list.print_sorted()
[-5, -3, 0, 10]

>>> my_list = MyList([5, 5, 5, 3, 3, 1])
>>> my_list.print_sorted()
[1, 3, 3, 5, 5, 5]

>>> my_list = MyList(['c', 'a', 'b'])
>>> my_list.print_sorted()
['a', 'b', 'c']

>>> my_list = MyList([3.5, 1.2, 2.8])
>>> my_list.print_sorted()
[1.2, 2.8, 3.5]

>>> my_list = MyList([4, 2, 6])
>>> my_list.print_sorted()
[2, 4, 6]
        
>>> my_list = MyList(['z', 'x', 'y'])
>>> my_list.print_sorted()
['x', 'y', 'z']

>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
